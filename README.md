# ChM_lab4
Лабораторна робота №4 з чисельних методів
Обчислює інтеграл функції користуючись правилом Рунге за формулою Сімпсона

Початково задані параметри:

a = -1.6514 # Лівий кінець інтегрування
b = 0.1514 # Правий кінець інтегрування
p = 4 # Степінь точності формули Сімпсона
eps = 1E-4 # Точність знаходження інтегралу
debug = False # Закоментувати, щоб зачити debug інформацію в консолі
maxIter = 1000 # Максимальна кількість ітерацій, після якої викинеться Exception
fname = "log.txt" # файл для виводу інформації про обчислення

executable file "main.py"
